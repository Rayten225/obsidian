### 🎡 Зацикленный револьвер квизов

```dataviewjs
const tFile = app.vault.getAbstractFileByPath(dv.current().file.path);
const allTasks = Array.from(dv.current().file.tasks.where(t => t.text.includes("http")));

if (allTasks.length === 0) {
    dv.paragraph("🔻 **Барабан пуст.** Скинь в самый низ файла ссылки в формате `- [ ] https://...`");
} else {
    const openTasks = allTasks.filter(t => !t.completed);

    if (openTasks.length > 0) {
        const current = openTasks[0];
        const uMatch = current.text.match(/https?:\/\/[^\s)]+/);
        const url = uMatch ? uMatch[0] : "#";
        
        const nameMatch = current.text.match(/\(([^)]+)\)/);
        const quizName = nameMatch ? nameMatch[1] : `Снаряд №${allTasks.indexOf(current) + 1}`;

        // --- 🚀 АКТИВНАЯ КНОПКА "ПРОЙТИ" (С АВТО-ЧЕКБОКСОМ) ---
        const takeBtn = document.createElement("button");
        takeBtn.innerHTML = `🚀 ПРОЙТИ: <b>${quizName}</b>`;
        takeBtn.style.cssText = "display: block; width: 100%; margin: 12px 0; padding: 14px 20px; background-color: #2563EB; color: white; border-radius: 8px; border: none; font-size: 16px; cursor: pointer; box-shadow: 0 4px 12px rgba(37, 99, 235, 0.3);";
        
        takeBtn.onclick = async () => {
            window.open(url, '_blank'); // Поток 1: Открываем Школково
            // Поток 2: Вскрываем файл и заменяем [ ] на [x] именно у этой строчки
            await app.vault.process(tFile, data => data.replace("- [ ] " + current.text, "- [x] " + current.text));
        };

        dv.container.appendChild(takeBtn);

        // Отрисовка хвоста очереди
        const carousel = [...allTasks].sort((a, b) => Number(a.completed) - Number(b.completed));

        dv.table(["Статус", "Тест"], 
            carousel.map(t => {
                const u = t.text.match(/https?:\/\/[^\s)]+/)?.[0] || "#";
                const nMatch = t.text.match(/\(([^)]+)\)/);
                const name = nMatch ? nMatch[1] : `Квиз #${allTasks.indexOf(t) + 1}`;
                
                let status = "⏳ в очереди";
                if (t.completed) status = `<span style="color:#10B981;">✅ пройден</span>`;
                if (t === current) status = "👉 <b>СЕЙЧАС</b>";

                return [status, `<a href="${u}">${name}</a>`];
            })
        );

    } else {
        // --- 🔄 КНОПКА "ПЕРЕЗАРЯДИТЬ" (АВТО-СБРОС ВСЕХ ГАЛОЧЕК) ---
        const resetBtn = document.createElement("button");
        resetBtn.innerHTML = "🔄 <b>ПЕРЕЗАРЯДИТЬ БАРАБАН</b><br><span style='font-size:12px; opacity: 0.8;'>Сбросить 100% галочек и пойти на следующий круг</span>";
        resetBtn.style.cssText = "display: block; width: 100%; margin: 16px 0; padding: 16px; background-color: #10B981; color: white; border-radius: 8px; border: none; font-size: 16px; cursor: pointer; box-shadow: 0 4px 12px rgba(16, 185, 129, 0.3);";
        
        resetBtn.onclick = async () => {
            // Вскрываем файл и сносим галочки ТОЛЬКО у веб-ссылок
            await app.vault.process(tFile, data => data.replace(/- \[x\] (.*http.*)/g, "- [ ] $1"));
        };

        dv.container.appendChild(resetBtn);
    }
}
```
